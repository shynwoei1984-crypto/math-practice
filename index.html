function generateGeometryQuestions() { const list = []; const units = ["cm", "m"]; for (let i = 1; i <= 10; i++) { let label = "", value = 0, unit = "", svg = "", hint = ""; let difficultyDesc = i <= 5 ? "【基本題】" : "【挑戰題】"; let u = units[Math.floor(Math.random() * units.length)]; let uArea = u + "²"; if (i === 1) { // Q1: Rectangle Perimeter label = "求出此長方形的「周長」"; let w = Math.floor(Math.random() * 6) + 6; } else if (i === 3) { // Q3: Rectangle Area label = "求出此長方形的「面積」"; let w = Math.floor(Math.random() * 6) + 8; // 8~13 let h = Math.floor(Math.random() * 4) + 3; // 3~6 value = w * h; unit = uArea; svg = ``; hint = `${difficultyDesc}提示：長方形面積公式 = $\\text{長} \\times \\text{寬}$。`; } else if (i === 4) { // Q4: Reverse - Given Perimeter of Square, find side length let s = Math.floor(Math.random() * 6) + 4; // 4~9 let p = s * 4;
